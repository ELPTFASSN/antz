106  //
107  // Add a line to a circular buffer of text lines
108  //
109  // Wrap the buffer destination when end is reached.
110  //
111  // If the current read location overlaps the area being written, increment it until it doesn't
112  //
113  int counter;
114  void AddLine( const char* line )
115  {
116  counter++;
117  //printf("AddLine %d: %s\n", counter, line );
118  char* linePtrStart = &lineData[ nextWriteBufferIdx ];
119  nextWriteBufferIdx += strlen(line);
120  nextWriteBufferIdx++;
121  char* linePtrEnd = &lineData[ nextWriteBufferIdx ];
122  if (nextWriteBufferIdx > LINE_WRAP_THRESHHOLD) {
123  nextWriteBufferIdx = 0;
124  }
125 
126  // If we have already read something, and this is going to wipe out some data, adjust the pointers to ignore the data being
127  // wiped out
128  if (nextReadLineIdx != -1) {
129  while ((lines[nextReadLineIdx] >= linePtrStart) && (lines[nextReadLineIdx] <= linePtrEnd)) {
130  nextReadLineIdx++;
131  if (nextReadLineIdx >= NUMBER_LINES) {
132  nextReadLineIdx = 0;
133  }
134  }
135  }
136 
137  //
138  // Copy the line, update the line pointers
139  //
140  strcpy( linePtrStart, line );
141  lines[nextWriteLineIdx++] = linePtrStart;
142  if (nextWriteLineIdx >= NUMBER_LINES) {
143  nextWriteLineIdx = 0;
144  }
145  // if there has never been a read index, it gets initialized to 0 (since there is now data to read)
146  if (nextReadLineIdx == -1) {
147  nextReadLineIdx = 0;
148  // if the readindex is the same as the write index
149  } else if (nextReadLineIdx == nextWriteLineIdx) {
150  nextReadLineIdx++;
151  if (nextReadLineIdx >= NUMBER_LINES) {
152  nextReadLineIdx = 0;
153  }
154  }
155  dump();
156  }
