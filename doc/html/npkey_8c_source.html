46 /*------------------------------------------------------------------------------
47 * --- KEY ARCHITECTURE PLAN ---
48 *
49 * process both the OS keyboard API (low-latency) and HID devices (slower)
50 * HID allows support for multiple independent devices which are then
51 * mapped to the active context(s) as either ASCII text or RAW keys.
52 *
53 * uses include action keys, quick keys, typing and editing with dynamic-suggest
54 *
55 * allow for multiple keyboards to have independent target contexts to allow
56 * a multi-user(s) to have separate consoles and/or other active contexts.
57 *
58 * event driven model
59 * first sets modifiers states
60 * then converts API key code to ASCII or RAW
61 * next call GUI/\console event handlers that are registered for the device(s)
62 *
63 * startup...
64 * console->mode = kNPconsoleCmd; //default keyboard context is rootHUD
65 * keyboard[i]->context = rootHUD //root HUD console
66 *
67 * active window is called (currently handled by GLUT)
68 * keyboard (and mouse) updates based on active context
69 * key/mouse (KM) pair assigned together with unique selection set
70 * sub-selection / context within the window
71 * ie. control two separate consoles
72 *
73 * focus within the window is directed to the devices active context
74 * rootHUD console and other node consoles... anywhere in tree
75 *
76 * event handler(s) called in order until event is handled
77 * first handler could be ASCII Viewer user input while next is 3D Viewer, etc
78 * allow for multiple simultaneous methods of user feedback, DB tree viewer
79 *
80 * currently supports a standard US English QWERTY keyboard
81 * allow user re-mapping for non-QWERTY layout and non-English keyboards
82 * API keyboard support for ASCII, GLUT, OSX, MSW, Linux, HID
83 * the two fundamental types are ASCII (typewriter style) and Game Mode
84 * gamemode = kNPkeyRAW or edit interface uses kNPkeyASCII + special / modifiers
85 * keyEventType = kNPkeyDown //Up Repeat
86 * keyType = QWERTY / ASCII / modifier / special
87 * other HID devices, languages...?
88 * keyRepeatRate = 15
89 * optionally load keymaps from CSV
90 * default translate ASCII to local ASCII-friendly key code list
91 *
92 * Cross-Platform issues include the Apple 'Command' key and how the user
93 * expects the ctrl, alt and command key to behave, such as copy paste...
94 * also, ALT key only generates an event when another key gets pressed
95 * releasing SHIFT key stops a key repeat, pressing it down does not...
96 * so... CapsLock, Shift, Alt and Command are all a little different!
97 *
98 *----------------------------------------------------------------------------- */

359 void npKeyEvent (int key, int keyEventType)
360 {
361  int command = 0;
362 
363  pData data = (pData) npGetDataRef();
364 
365  if (key == -1) //unknown key type
366  return;
367 
368  if ( data->io.gl.hud.console.mode > 0 )
369  npConsoleKeyEvent (key, keyEventType, data);
370  else
371  {
372  command = data->io.key.map[keyEventType][key];
373 
374  if (command != 0)
375  npCtrlCommand (command, data);
376  }
377 }

585 //------------------------------------------------------------------------------
586 //Process key events from GLUT, both ASCII and 'Special' keys and modifiers
587 //note that freeglut supports shift, alt, and ctrl but not the OSX 'command' key
588 //
589 //there are two distinct fundamental keyboard modes of operation
590 //either ASCII text mode 'typewriter style' or raw key events 'game mode'
591 //
592 //game mode treats every key as a RAW switch, no difference between 'a' and 'A'
593 //caps lock key is unique in that it only generates key down events
594 //all other keys have up and down events as well as L vs R modifier distinction
595 //note that modifier key behavior varies depending on the OS and library
596 //
597 //freeglut mixes a modifier key up event as an ASCII event 16, 17 or 18
598 //it generates 'Special' key up and down events 112-117 for both L/R modifiers
599 //then follows with an additional ASCII key up, 16(Shift), 17(Ctrl) and 18(Alt)
600 //
601 //key handling is often confusing, inconsistent and un-documented across OS APIs
602 //try printing raw event values and testing all keys when implementing a new API
603 //------------------------------------------------------------------------------
604 void npKeyGlut (int key, int x, int y, int keyTypeGlut, int modifiers)
605 {
606  int keyMode = 0;
607 
608  pData data = (pData) npGetDataRef();
609 
610  pNPkey keyboard = &data->io.key;
611 
612 
613 /* if (modifiers == GLUT_ACTIVE_SHIFT) //no longer using this method
614  data->io.key.modShift = true;
615  else
616  data->io.key.modShift = false;
617  if (modifiers == GLUT_ACTIVE_CTRL)
618  data->io.key.modCtrl = true;
619  else
620  data->io.key.modCtrl = false;
621  if (modifiers == GLUT_ACTIVE_ALT)
622  data->io.key.modAlt = true;
623  else
624  data->io.key.modAlt = false;
625 */
626  //first catch and calculate the modifier key states
627  //zz debug, far as i know... this is un-documented freeglut behavior
628  //needs testing, may not work with Apple GLUT or other OS
629  //typically would call glutGetModifiers() instead of this method
630  //this will take care of returning the states in case of a missed key up
631  switch (keyTypeGlut) {
632  case kGlutKeyUp : //note that these are ONLY key up events
633  switch (key) {
634  case 16 : //both SHIFT keys released
635  keyboard->modShift = false;
636  keyboard->modShiftLeft = false;
637  keyboard->modShiftRight = false;
638  break;
639  case 17 : //both CTRL keys released
640  keyboard->modCtrl = false;
641  keyboard->modCtrlLeft = false;
642  keyboard->modCtrlRight = false;
643  break;
644  case 18 : //both ALT keys released
645  keyboard->modAlt = false;
646  keyboard->modAltLeft = false;
647  keyboard->modAltRight = false;
648  break;
649  //initial caps states needs to be detected //zz debug
650  case 20 : //toggle stored caps state
651  keyboard->modCapsLock = 1 - keyboard->modCapsLock;
652  break;
653  default : break;
654  }
655  break;
656  case kGlutKeyDownSpecial: // GLUT key codes for non-ASCII keys
657  switch (key) {
658  case 112 : keyboard->modShiftLeft = true; break;
659  case 113 : keyboard->modShiftRight = true; break;
660  case 114 : keyboard->modCtrlLeft = true; break;
661  case 115 : keyboard->modCtrlRight = true; break;
662  case 116 : keyboard->modAltLeft = true; break;
663  case 117 : keyboard->modAltRight = true; break;
664  default : break;
665  }
666  break;
667  case kGlutKeyUpSpecial: // non-ASCII Glut constants
668  switch (key) {
669  case 112 : keyboard->modShiftLeft = false; break;
670  case 113 : keyboard->modShiftRight = false; break;
671  case 114 : keyboard->modCtrlLeft = false; break;
672  case 115 : keyboard->modCtrlRight = false; break;
673  case 116 : keyboard->modAltLeft = false; break;
674  case 117 : keyboard->modAltRight = false; break;
675  default : break;
676  }
677  break;
678  default : break;
679  }
680 
681  //calculate sum states
682  if (keyboard->modShiftLeft || keyboard->modShiftRight)
683  keyboard->modShift = true;
684  else
685  keyboard->modShift = false;
686 
687  if (keyboard->modAltLeft || keyboard->modAltRight)
688  keyboard->modAlt = true;
689  else
690  keyboard->modAlt = false;
691 
692  if (keyboard->modCtrlLeft || keyboard->modCtrlRight)
693  keyboard->modCtrl = true;
694  else
695  keyboard->modCtrl = false;
696 
697  if (keyboard->modShift || keyboard->modCapsLock)
698  {
699  if (keyboard->modShift && keyboard->modCapsLock)
700  keyboard->modAlphaUpper = false;
701  else
702  keyboard->modAlphaUpper = true;
703  }
704  else
705  keyboard->modAlphaUpper = false;
706 
707  //printf("caps: %d shift: %d upper: %d alt: %d ctrl: %d \n",
708  // keyboard->modCapsLock, keyboard->modShift, keyboard->modAlphaUpper,
709  // keyboard->modAlt, keyboard->modCtrl);
710 
711  //set the key process mode, either raw keys or ASCII typewriter style
712  if( data->io.gl.hud.console.mode > 0 )
713  keyMode = kNPkeyModeASCII;
714  else
715  keyMode = kNPkeyModeRAW;
716 
717  //call the relevant key handler based on the target mode
718  switch (keyMode) {
719  case kNPkeyModeRAW :
720  switch (keyTypeGlut) {
721  case kGlutKeyDown: // ASCII key codes converted to native
722  npKeyEvent (npKeyRAW(key, kNPkeyCodeASCII), kKeyDown);
723  break;
724  case kGlutKeyUp: // ASCII key codes converted
725  npKeyEvent (npKeyRAW(key, kNPkeyCodeASCII), kKeyUp);
726  break;
727  case kGlutKeyDownSpecial: // GLUT key codes for non-ASCII keys
728  npKeyEvent (npKeyRAW(key, kNPkeyCodeGLUTspecial), kKeyDown);
729  break;
730  case kGlutKeyUpSpecial: // non-ASCII Glut constants
731  npKeyEvent (npKeyRAW(key, kNPkeyCodeGLUTspecial), kKeyUp);
732  break;
733  default : break;
734  }
735  break;
736  case kNPkeyModeASCII : //send raw ASCII text to console to process
737  switch (keyTypeGlut) {
738  case kGlutKeyDown: npConsoleKeyEvent (key, kKeyDownASCII, data);
739  break;
740  case kGlutKeyDownSpecial: // non-ASCII Glut constants
741  switch (key) {
742  case GLUT_KEY_LEFT :
743  npConsoleKeyEvent (kKeyCodeLeft, kKeyDownSpecial, data);
744  break;
745  case GLUT_KEY_RIGHT :
746  npConsoleKeyEvent (kKeyCodeRight, kKeyDownSpecial, data);
747  break;
748  case GLUT_KEY_UP :
749  npConsoleKeyEvent (kKeyCodeUp, kKeyDownSpecial, data);
750  break;
751  case GLUT_KEY_DOWN :
752  npConsoleKeyEvent (kKeyCodeDown, kKeyDownSpecial, data);
753  break;
754  default : break;
755  }
756  break;
757  default : break;
758  }
759  default : break;
760  }
761 }
