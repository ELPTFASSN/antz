362 void npPostNodeID( pNPnode node, void* dataRef )
363 {
364  pData data = (pData) dataRef;
365 
366  char msg[256];
367  char msgPart[256];
368 
369  msg[0] = '\0';
370  msgPart[0] = '\0';
371 
372  if( node->type == kNodeLink )
373  strcpy( msg, "edge " );
374  else if( node->branchLevel && !node->childCount )
375  strcpy( msg, "leaf " );
376  else if( !node->childCount )
377  strcpy( msg, "solo " ); //tree with only root node
378  else if ( !node->branchLevel )
379  strcpy( msg, "root " );
380  else
381  strcpy( msg, "nexus " ); //internal node that is neither root nor leaf
382 
383 
384  if (node->type == kNodeCamera)
385  sprintf(msgPart, "%s cam ", msg);
386  if (node->type == kNodeGrid)
387  sprintf(msgPart, "%s grid", msg);
388  if (node->type == kNodePin)
389  sprintf(msgPart, "%s pin ", msg);
390  if (node->type == kNodeLink)
391  sprintf(msgPart, "%s link", msg);
392 
393  //display id level and tag //zz add subspace grid_id?
394  if (node->recordID)
395  sprintf( msg, "%s level: %-3d id: %-6d rec: %-6d tag: %s", msgPart,
396  node->branchLevel + 1, node->id, node->recordID, node->tag->title );
397  else
398  sprintf( msg, "%s level: %-3d id: %-6d", msgPart,
399  node->branchLevel + 1, node->id );
400 
401  npPostMsg (msg, kNPmsgCtrl, dataRef);
402 }
