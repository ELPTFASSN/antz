166 {
167  if (node->freeze) //do not update if frozen
168  return;
169 
170  switch (node->type) //types organized to optimize switch statement speed

178  case kNodePoints : UpdateNodePoints (node, dataRef); break;
179  case kNodeGrid : UpdateNodeGrid (node, dataRef); break;

372 void npCameraTranslate (void* nodeRef, void* dataRef )
373 {
374  pData data = (pData) dataRef;
375  pNPnode cam = (pNPnode) nodeRef;
376  pNPnode targetNode = data->map.selectedPinNode; //node[data->map.selectedPinIndex];
377  pNPnode rootGrid = data->map.node[kNPnodeRootGrid];
378 
379  pNPmouse mouse = &data->io.mouse;
380 
381  float fps = 60.0f;
382  float radius = 0.0f;
383  float cdX = 0.0f, cdY = 0.0f, cdZ = 0.0f;
384  float tX = 0.0f, tY = 0.0f, tZ = 0.0f;
385  float dX = 0.0f, dY = 0.0f, dZ = 0.0f;
386 
387  if (data->io.mouse.pickMode == kNPmodeGrid) //zz debug
388  targetNode = data->map.selectedGrid;
389 
390  //global position of entire scene is relative to the root grid scale.z
391  //cam target factors in root grid scale.z and pin scale.z times pin height
392  if (targetNode != NULL)
393  {
394  //traverse tree from root to calculate target world coordinates
395  // npNodeUpdateWorld (targetNode, data);
396 
397  mouse->targetDest.x = targetNode->world.x; //zz debug world position is 1-2w fraes behind
398  mouse->targetDest.y = targetNode->world.y; // calculated last draw time using GL matrix stack
399  mouse->targetDest.z = targetNode->world.z; // should calculate cam and target node position after all other nodes updated
400  }
401 
402  if (data->io.mouse.camMode == kNPmouseModeCamExamXZ)
403  cam->proximity.x *= 1.0f + data->io.mouse.delta.y * 0.005f;
404 
405  // set target based on radius, KML compliant, invert and 90 deg offset for rotate.x
406  cdX = -sinf(cam->rotate.y / kRADtoDEG) * cam->proximity.x * cosf((cam->rotate.x - 90.0f) / kRADtoDEG);
407  cdY = -cosf(cam->rotate.y / kRADtoDEG) * cam->proximity.x * cosf((cam->rotate.x - 90.0f) / kRADtoDEG);
408  cdZ = sinf((-90.0f - cam->rotate.x) / -kRADtoDEG) * cam->proximity.x;
409 
410  data->io.mouse.target.x = npLowPassFilter ( data->io.mouse.target.x, data->io.mouse.targetDest.x, 0.05f );
411  data->io.mouse.target.y = npLowPassFilter ( data->io.mouse.target.y, data->io.mouse.targetDest.y, 0.05f );
412  data->io.mouse.target.z = npLowPassFilter ( data->io.mouse.target.z, data->io.mouse.targetDest.z, 0.05f );
413 
414  tX = cdX + data->io.mouse.targetDest.x;
415  tY = cdY + data->io.mouse.targetDest.y;
416  tZ = cdZ + data->io.mouse.targetDest.z;
417 // tX = cdX + data->io.mouse.target.x;
418 // tY = cdY + data->io.mouse.target.y;
419 // tZ = cdZ + data->io.mouse.target.z;
420 
421  dX = cam->translate.x - tX;
422  dY = cam->translate.y - tY;
423  dZ = cam->translate.z - tZ;
424 
425  radius = sqrtf (dX*dX+dY*dY+dZ*dZ);
426  if (radius < 0.1f) //clamp
427  data->io.mouse.targeting = false;
428 
429  if (data->io.mouse.targeting)
430  {
431  //time corrected camera translation //zzhp
432  //could smooth out the starting trajectory, fps halfs when picking object //zz debug
433  fps = (float) (1.0 / data->io.cycleDelta);
434  cam->translate.x = npLowPassFilter ( cam->translate.x, tX, 0.05f * (65.0f / (fps + 5.0f)) );
435  cam->translate.y = npLowPassFilter ( cam->translate.y, tY, 0.05f * (65.0f / (fps + 5.0f)) );
436  cam->translate.z = npLowPassFilter ( cam->translate.z, tZ, 0.05f * (65.0f / (fps + 5.0f)) );
437  }
438  else
439  {
440  cam->translate.x = tX;
441  cam->translate.y = tY;
442  cam->translate.z = tZ;
443  }
444 }

451

462

469

471  cosY = (float) cos(node->rotate.y / kRADtoDEG);
472

474  node->translate.y += mouse->cmDY * cosY;
475  break;
476 
477  case kNPmouseModeCamFlyB : break;
478 
479  case kNPmouseModeDragXZ : break;
480  case kNPmouseModeDragXY : break;
481  case kNPmouseModeDragWorld : break;
482 
483  case kNPmouseModeCamExamXZ :
484  node->rotate.y += mouse->delta.x * 0.2f;
485  break;
486  case kNPmouseModeCamExamXY :
487  node->rotate.x += mouse->delta.y * -0.2f; //invert axis
