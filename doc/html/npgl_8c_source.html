204  glDepthFunc (GL_LESS);

207  {
208  glDisable (GL_LIGHTING); //disable lighting and alpha for pickPass

238  // glBlendFunc(GL_ONE_MINUS_DST_COLOR, GL_ONE_MINUS_SRC_COLOR);
239  // break;
240  default :
241  glBlendFunc (GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA); //MB-Transp
242  break;
243  }
244 
245  glEnable (GL_COLOR_MATERIAL); //without everything is grayscale
246 
247  glEnable (GL_LIGHT0);
248  glEnable (GL_LIGHT1);
249 
250  glLightfv (GL_LIGHT0, GL_POSITION, positionA);
251  glLightfv (GL_LIGHT0, GL_SPECULAR, white);
252  glLightfv (GL_LIGHT0, GL_AMBIENT, black);
253  glLightfv (GL_LIGHT0, GL_DIFFUSE, white);
254 
255  glLightfv (GL_LIGHT1, GL_POSITION, positionB);
256  glLightfv (GL_LIGHT1, GL_SPECULAR, white);
257  glLightfv (GL_LIGHT1, GL_AMBIENT, black);
258  glLightfv (GL_LIGHT1, GL_DIFFUSE, white);
259 }

407 {
408  int i = 0;
409  bool allSubNodesHidden = true;
410  pNPnode child = NULL;
411 
412  //normally clicked node should not be hidden, but just in case...
413  if (!hide)
414  node->hide = false;
415 
416  for (i=0; i < node->childCount; i++)
417  {
418  child = node->child[i];
419 
420  //picked node should be hidden only if all sub nodes already hidden
421  if (!child->hide)
422  allSubNodesHidden = false;
423 
424  // recursively call this function
425  npHideSubNodes (hide, child, dataRef);
426 
427  // apply passed in function operation to node
428  child->hide = hide;
429  }
430 
431  //picked node should be hidden only if all sub nodes already hidden
432  if (hide && allSubNodesHidden)
433  node->hide = true;
434  else
435  node->hide = false;
436 }
